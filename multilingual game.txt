    let currentLang = "en";
    let currentQuestions = [...QUESTIONS_EN];
    let traineeName = "";
    let roundActive = false;
    let currentIndex = 0;
    let score = 0;
    let answered = false;
    let selectedOpt = null;
    let roundQuestions = [];

    const root = document.getElementById("gameRoot");

    function getLangText(key) { return TRANSLATIONS[currentLang][key] || key; }

    function renderNameScreen() {
        root.innerHTML = `
            <div class="game-container">
                <div class="game-header">
                    <div class="game-title"><h2>${getLangText("appTitle")}</h2><div class="credit-line">${getLangText("credit")}</div></div>
                    <div class="lang-selector">
                        <button class="lang-btn ${currentLang === "en" ? "active" : ""}" data-lang="en">🇬🇧 EN</button>
                        <button class="lang-btn ${currentLang === "nl" ? "active" : ""}" data-lang="nl">🇳🇱 NL</button>
                        <button class="lang-btn ${currentLang === "fr" ? "active" : ""}" data-lang="fr">🇫🇷 FR</button>
                    </div>
                </div>
                <div class="game-body name-screen">
                    <h3>${getLangText("enterName")}</h3>
                    <div class="name-input-group">
                        <input type="text" id="traineeNameInput" placeholder="${getLangText("namePlaceholder")}">
                        <button class="start-btn" id="startGameBtn">${getLangText("startBtn")}</button>
                    </div>
                    <div class="game-footer">🔴 Red Card • 🟡 Yellow Card • 🔵 Blue Card • Workflow • Phrasebook</div>
                </div>
            </div>
        `;
        document.querySelectorAll(".lang-btn").forEach(btn => {
            btn.addEventListener("click", (e) => {
                currentLang = btn.dataset.lang;
                if (currentLang === "en") currentQuestions = [...QUESTIONS_EN];
                else if (currentLang === "nl") currentQuestions = [...QUESTIONS_NL];
                else currentQuestions = [...QUESTIONS_FR];
                renderNameScreen();
            });
        });
        document.getElementById("startGameBtn").addEventListener("click", () => {
            let name = document.getElementById("traineeNameInput").value.trim();
            traineeName = name || "Trainee";
            roundQuestions = [...currentQuestions];
            for (let i = roundQuestions.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [roundQuestions[i], roundQuestions[j]] = [roundQuestions[j], roundQuestions[i]];
            }
            currentIndex = 0; score = 0; answered = false; selectedOpt = null; roundActive = true;
            renderQuestion();
        });
    }

    function renderQuestion() {
        if (!roundActive || currentIndex >= roundQuestions.length) { finishRound(); return; }
        const q = roundQuestions[currentIndex];
        const optsHtml = q.options.map((opt, idx) => `<button class="option-btn" data-opt="${idx}">${String.fromCharCode(65+idx)}. ${escapeHtml(opt)}</button>`).join('');
        root.innerHTML = `
            <div class="game-container">
                <div class="game-header">
                    <div class="game-title"><h2>${getLangText("appTitle")}</h2><div class="credit-line">${getLangText("credit")}</div></div>
                    <div class="trainee-badge">${getLangText("traineeLabel")} ${escapeHtml(traineeName)}</div>
                    <div class="score-board">⭐ ${score}/${currentIndex}</div>
                </div>
                <div class="game-body">
                    <div class="scenario-card">📌 <strong>${currentIndex+1}/${roundQuestions.length}</strong><br>${escapeHtml(q.text)}</div>
                    <div class="options-list" id="optsList">${optsHtml}</div>
                    <div id="fbArea" class="feedback-panel" style="display:none;"></div>
                    <button class="btn-next" id="nextBtn" disabled>${getLangText("submitBtn")}</button>
                    <div class="game-footer">🔍 Select an answer, then validate.</div>
                </div>
            </div>
        `;
        let selected = null;
        document.querySelectorAll(".option-btn").forEach(btn => {
            btn.addEventListener("click", () => {
                if (answered) return;
                document.querySelectorAll(".option-btn").forEach(b => { b.style.background = "#f0f2f5"; });
                btn.style.background = "#c1e0ff";
                selected = parseInt(btn.dataset.opt);
                document.getElementById("nextBtn").disabled = false;
            });
        });
        document.getElementById("nextBtn").addEventListener("click", () => {
            if (answered || selected === undefined) return;
            const qq = roundQuestions[currentIndex];
            const isCorrect = (selected === qq.correct);
            if (isCorrect) score++;
            const fbDiv = document.getElementById("fbArea");
            const correctLetter = String.fromCharCode(65+qq.correct);
            if (isCorrect) {
                fbDiv.innerHTML = `<strong>${getLangText("correctFeedback")}</strong><br>${escapeHtml(qq.explanation)}<div class="doc-ref">📄 ${escapeHtml(qq.docRef)}</div>`;
                fbDiv.className = "feedback-panel feedback-correct";
            } else {
                fbDiv.innerHTML = `<strong>${getLangText("wrongFeedback")}</strong><br>${getLangText("yourChoice")}: ${String.fromCharCode(65+selected)}. ${escapeHtml(qq.options[selected])}<br>${getLangText("correctAnswer")}: ${correctLetter}. ${escapeHtml(qq.options[qq.correct])}<br><br>${escapeHtml(qq.explanation)}<div class="doc-ref">📄 ${escapeHtml(qq.docRef)}</div>`;
                fbDiv.className = "feedback-panel feedback-wrong";
            }
            fbDiv.style.display = "block";
            document.querySelectorAll(".option-btn").forEach((btn, idx) => {
                btn.disabled = true;
                if (idx === qq.correct) { btn.style.background = "#c8f0d1"; }
                if (!isCorrect && idx === selected && idx !== qq.correct) { btn.style.background = "#ffdddd"; }
            });
            answered = true;
            const nextBtn = document.getElementById("nextBtn");
            nextBtn.innerText = (currentIndex+1 === roundQuestions.length) ? getLangText("finishBtn") : getLangText("nextBtn");
            const newNext = nextBtn.cloneNode(true);
            nextBtn.parentNode.replaceChild(newNext, nextBtn);
            newNext.addEventListener("click", () => {
                if (currentIndex+1 >= roundQuestions.length) finishRound();
                else { currentIndex++; answered = false; selected = undefined; renderQuestion(); }
            });
        });
    }

    function finishRound() {
        const total = roundQuestions.length;
        root.innerHTML = `
            <div class="game-container">
                <div class="game-header"><div class="game-title"><h2>${getLangText("appTitle")}</h2><div class="credit-line">${getLangText("credit")}</div></div></div>
                <div class="game-body result-area">
                    <div class="score-summary">🎯 <strong>${escapeHtml(traineeName)}</strong> ${getLangText("finalScoreMsg")} <strong style="font-size:2rem;">${score}</strong> ${getLangText("upTo")} ${total}.<br>(${score}/${total} · ${Math.round(score/total*100)}%)<br>${score >= 15 ? getLangText("passMsg") : getLangText("failMsg")}</div>
                    <button class="restart-btn" id="restartBtn">${getLangText("newRound")}</button>
                </div>
                <div class="game-footer">JUNIOR QC ANALYST • multilingual</div>
            </div>
        `;
        document.getElementById("restartBtn").addEventListener("click", () => { roundActive = false; renderNameScreen(); });
    }

    function escapeHtml(str) { if (!str) return ""; return str.replace(/[&<>]/g, m => ({ '&':'&amp;', '<':'&lt;', '>':'&gt;' }[m])); }
